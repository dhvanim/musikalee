192def follower_update_db(user): 

193 """ 

194 Get List of followers 

195 """ 

196 followers_list = DB.session.query( 

197 models.Users.following).filter_by( 

198 username=user).scalar() 

199 new_followers_list = followers_list 

200 DB.session.commit() 

201 if user not in followers_list: 

202 new_followers_list.append(user) 

203 is_followed = True 

204 DB.session.query(models.Users).filter( 

205 models.Users.username == user).update( 

206 {models.Users.following: new_followers_list}, 

207 synchronize_session="fetch") 

208 else: 

209 new_followers_list.remove(user) 

210 is_followed = False

212 models.Users.username == user).update( 

213 {models.Users.following: new_followers_list}, 

214 synchronize_session="fetch") 

215 DB.session.commit() 

216 return [new_followers_list, is_followed]
